# 主从复制

# 配置原理

主 Redis 配置：无需配置

从 Redis 配置：

修改从服务器上的 redis.conf 文件

# slaveof <masterip> <masterport> 
# 表示当前【从服务器】对应的【主服务器】的IP是192.168.10.135，端口是6379。 
replicaof 127.0.0.1 6379

第一步：安装 C 语言需要的 GCC 环境，并更新到 9 版本以上，不然会编译报错

yum install -y gcc-c++ 
yum install -y wget
yum -y install centos-release-scl
#临时设置 gcc 版本
scl enable devtoolset-9 bash
#永久设置 gcc 版本
echo "source /opt/rh/devtoolset-9/enable" >>/etc/profile


#第二步：下载并解压缩 Redis 源码压缩包
wget https://download.redis.io/releases/redis-6.0.10.tar.gz
tar -zxf redis-6.0.10.tar.gz
#第三步：编译 Redis 源码，进入 redis-6.0.10 目录，执行编译命令
cd redis-6.0.10/src 
make
make test(不必须)

编译完成后

如果 make test 失败提醒需要更新 tcl 版本

wget http://downloads.sourceforge.net/tcl/tcl8.6.1-src.tar.gz  
tar -zxf tcl8.6.1-src.tar.gz
cd  tcl8.6.1/unix/  
sudo ./configure  
sudo make  
sudo make install

第四步：安装 Redis，需要通过 PREFIX 指定安装路径

mkdir /usr/local/env/redis -p 
make install PREFIX=/usr/local/env/redis
#复制conf
cd ..
cp redis.conf /usr/local/env/redis/bin/
cp sentinel.conf /usr/local/env/redis/bin/

#修改 redis.conf

cd ..
cd redis/bin
vim redis.conf

#注释掉则不限于本机器访问
bind 127.0.0.1
#no 允许外界访问
protected-mode no
#yes 守护进程后台启动
daemonize yes

#占用窗口启动 --不可以被远程访问
./redis-server
#关闭
control + c
#不占用窗口启动 --不可以被远程访问
./redis-server & 
#守护进程后台启动 --可以被远程访问
 ./redis-server redis.conf
#守护进程后台启动关闭ll
./redis-cli shutdown  
#查看进程
ps -ef |grep redis
root      25498      1  0 09:56 ?        00:00:00 ./redis-server 127.0.0.1:6379
#关闭进程
kill -9 25498
#客户端启动
./redis-cli -h 127.0.0.1 -p 6379

# 搭建主从

cd /usr/local/env
mkdir redis-ms
cp -r redis redis-ms/redis-master
cp -r redis redis-ms/redis-slave1
cp -r redis redis-ms/redis-slave2

修改 redis-slave1 的 redis.conf 内容为：

replicaof 127.0.0.1 6380
port 6381（大约在地91行左右）

修改 redis-slave2 的 redis.conf 内容为：

replicaof 127.0.0.1 6380
port 6382

分别启动三台 redis 服务器

编写启动脚本 start.sh 和关闭脚本 shutdown.sh (这里后三行的 sentinel 脚本先不要复制进去)

/usr/local/env/redis-ms/redis-master/bin/redis-server /usr/local/env/redis-ms/redis-master/bin/redis.conf
/usr/local/env/redis-ms/redis-slave1/bin/redis-server /usr/local/env/redis-ms/redis-slave1/bin/redis.conf
/usr/local/env/redis-ms/redis-slave2/bin/redis-server /usr/local/env/redis-ms/redis-slave2/bin/redis.conf

/usr/local/env/redis-ms/redis-master/bin/redis-cli -p 6380 shutdown
/usr/local/env/redis-ms/redis-slave1/bin/redis-cli -p 6381 shutdown
/usr/local/env/redis-ms/redis-slave2/bin/redis-cli -p 6382 shutdown

赋予权限

chmod +x start.sh shutdown.sh

切换到 master

./redis-master/bin/redis-cli -p 6380

切换到 redis-slave1

./redis-master/bin/redis-cli -p 6381

可以看到 从服务器已经自动复制了主服务器的数据，并且无法写

选举主服务器实验

查看主服务器进程号，然后杀死

从服务器无法选举出主服务器

# 总结

主服务可以读写，从服务器只能读。并且主服务器挂了之后，从服务器无法选举为主服务器。

# sentinel 模式

  1. 修改 sentinel.conf
port 26380

daemonize yes

sentinel monitor mymaster 127.0.0.1 6380 2
  1. 新建 start-sentinel.sh 和 shutdown-sentinel.sh 脚本
/usr/local/env/redis-ms/redis-master/bin/redis-server /usr/local/env/redis-ms/redis-master/bin/redis.conf
/usr/local/env/redis-ms/redis-slave1/bin/redis-server /usr/local/env/redis-ms/redis-slave1/bin/redis.conf
/usr/local/env/redis-ms/redis-slave2/bin/redis-server /usr/local/env/redis-ms/redis-slave2/bin/redis.conf
/usr/local/env/redis-ms/redis-master/bin/redis-sentinel  /usr/local/env/redis-ms/redis-master/bin/sentinel.conf
/usr/local/env/redis-ms/redis-slave1/bin/redis-sentinel  /usr/local/env/redis-ms/redis-slave1/bin/sentinel.conf
/usr/local/env/redis-ms/redis-slave2/bin/redis-sentinel  /usr/local/env/redis-ms/redis-slave2/bin/sentinel.conf

/usr/local/env/redis-ms/redis-master/bin/redis-cli -p 6380 shutdown
/usr/local/env/redis-ms/redis-slave1/bin/redis-cli -p 6381 shutdown
/usr/local/env/redis-ms/redis-slave2/bin/redis-cli -p 6382 shutdown
/usr/local/env/redis-ms/redis-master/bin/redis-cli -p 26380 shutdown
/usr/local/env/redis-ms/redis-slave1/bin/redis-cli -p 26381 shutdown
/usr/local/env/redis-ms/redis-slave2/bin/redis-cli -p 26382 shutdown

并赋予权限

chmod +x start-sentinel.sh shutdown-sentinel.sh
  1. 测试
./redis-master/bin/redis-cli -p 26380

查看 master 节点和 slave 节点信息

127.0.0.1:26380> sentinel master mymaster
127.0.0.1:26380> sentinel slaves mymaster
